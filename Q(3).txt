mysql> select e.Name as EmployeeName,
    -> e.Emp_city as EmployeeCity,
    -> c.Company_name as CompanyName,
    -> c.salary
    -> from Employee e
    -> join Company c
    -> on e.Emp_no = c.Emp_no
    -> where c.Salary > 10000;
